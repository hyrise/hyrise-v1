53 void PointerCalculator::updateFieldMapping() {
54  slice_for_slice.clear();
55  offset_in_slice.clear();
56  width_for_slice.clear();
57  slice_count = 0;
58 
59  size_t field = 0, dst_field = 0;
60  size_t last_field = 0;
61  bool last_field_set = false;
62 
63  for (size_t src_slice = 0; src_slice < table->sliceCount(); src_slice++) {
64  last_field_set = false;
65  for (size_t src_field = 0; src_field < table->getSliceWidth(src_slice) / sizeof(value_id_t); src_field++) {
66  // Check if we have to increase the fields until we reach
67  // a projected attribute
68  if (fields && (fields->size() <= dst_field || fields->at(dst_field) != field)) {
69  field++;
70  continue;
71  }
72 
73  if (!last_field_set || field > last_field + 1) { // new slice
74  slice_count++;
75  slice_for_slice.push_back(src_slice);
76  offset_in_slice.push_back(src_field);
77  width_for_slice.push_back(sizeof(value_id_t));
78  } else {
79  width_for_slice[slice_count - 1] += sizeof(value_id_t);
80  }
81 
82  dst_field++;
83  last_field = field;
84 
85  if (!last_field_set) {
86  last_field_set = true;
87  }
88 
89  field++;
90  }
91  }
92 }
