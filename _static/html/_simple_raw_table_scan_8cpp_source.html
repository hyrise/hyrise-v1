74  if (!table)

79  for(size_t i=0; i<table->columnCount(); ++i)

84  false, /* sorted */
85  64,

87  false /* compressed */);

93 
94  size_t tabSize = table->size();
95  for(size_t row=0; row < tabSize; ++row) {
96  if ((*_comparator)(row)) {

99  for(size_t i=0; i < result->columnCount(); ++i) {

107  }
