18 void NestedLoopEquiJoin::executePlanOperation() {
19  // get the two input tables and its hashmaps
20  // First part is the probe table
21  auto left = input.getTable(0);
22  const auto &hleft = input.getTable(1);
23  const auto &pleft = input.getTable(2);
24  const size_t left_size = left->size();
25 
26  // Now fetch the hash table
27  auto right = input.getTable(3);
28  const auto &hright = input.getTable(4);
29  const auto &pright = input.getTable(5);
30  const size_t right_size = right->size();
31 
32  // cast down left hash table, first column contains hashes (value_id_t), second column contains pos_t
33  const auto &lavs0 = hleft->getAttributeVectors(0);
34  const auto &lhvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(lavs0.at(0).attribute_vector);
35  const auto &lavs1 = hleft->getAttributeVectors(1);
36  const auto &lpvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(lavs1.at(0).attribute_vector);
37  // cast down left prefix sum table
38  const auto &lavsp = pleft->getAttributeVectors(0);
39  const auto &lprefixvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(lavsp.at(0).attribute_vector);
40  const size_t lprefixvector_size = lprefixvector->size();
41 
42  // cast down right hash table
43  const auto &ravs0 = hright->getAttributeVectors(0);
44  const auto &rhvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(ravs0.at(0).attribute_vector);
45  const auto &ravs1 = hright->getAttributeVectors(1);
46  const auto &rpvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(ravs1.at(0).attribute_vector);
47  // cast down right first prefix sum table
48  const auto &ravsp = pright->getAttributeVectors(0);
49  const auto &rprefixvector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(ravsp.at(0).attribute_vector);
50  const size_t rprefixvector_size = rprefixvector->size();
51 
52  // Prepare mask for right prefix sum
53  register auto mask1 = (1 << bits1()) - 1;
54  register auto mask2 = ((1 << bits2()) - 1) << bits1();
55  // join
56  size_t left_begin = 0, left_end = 0, right_begin = 0, right_end = 0, partition = 0, rpart = 0;
57  register uint32_t lhash;
58  auto lpos_list = new pos_list_t;
59  auto rpos_list = new pos_list_t;
60 
61  lpos_list->reserve(lhvector->size());
62  rpos_list->reserve(lhvector->size());
63 
64  register auto multiplier = 1 << bits2();
65  register auto shift_right = bits1();
66 
67  // iterate over partitions -> partition gives the offset into the prefix table
68  // The number of partitions depends on the number of bits used for the partitioning
69  for(size_t i = 0, partitions_count = _partitions.size(); i < partitions_count; i++){
70  partition = _partitions[i];
71 
72  // get the corresponding rows for partition using the left prefix table
73  left_begin = lprefixvector->get(0, partition);
74  if((partition == lprefixvector_size - 1)) //|| (i+1) == _partitions.size())
75  left_end = left_size;
76  else
77  left_end = lprefixvector->get(0,partition +1);
78 
79  // iterate over values in partition
80  for (size_t left_row = left_begin; left_row < left_end; left_row++) {
81  // find corresponding values of the right relation using the left hash value
82  lhash = lhvector->get(0, left_row);
83 
84  // determine partition in right prefix
85  rpart = ((lhash & mask1) * (multiplier)) + ((lhash & mask2) >> shift_right);
86  right_begin = rprefixvector->get(0, rpart);
87  right_end = rpart < (rprefixvector_size - 1) ? rprefixvector->get(0,rpart +1) : right_size;
88 
89  // iterate over all potentially matching positions in right relation
90  for (size_t right_row = right_begin; right_row < right_end; right_row++) {
91  // compare hashes
92  if (lhash == rhvector->get(0, right_row)) {
93  // remeber matching positions
94  lpos_list->push_back(lpvector->get(0, left_row));
95  rpos_list->push_back(rpvector->get(0, right_row));
96  }
97  }
98  }
99  }
100 
101  // if underlying table is pointer calc, get Actual Table as base for output pointer calc
102  const auto &rp = std::dynamic_pointer_cast<const PointerCalculator>(right);
103  const auto &lp = std::dynamic_pointer_cast<const PointerCalculator>(left);
104 
105  if(lp)
106  left = lp->getActualTable();
107  if(rp)
108  right = rp->getActualTable();
109 
110  // create PointerCalculator and pos_lists for output
111  auto loutput = PointerCalculatorFactory::createPointerCalculator(left, nullptr, lpos_list);
112  auto routput = PointerCalculatorFactory::createPointerCalculator(right, nullptr, rpos_list);
113 
114  // build output table
115  std::vector<storage::atable_ptr_t > vc;
116  vc.push_back(loutput);
117  vc.push_back(routput);
118 
119  storage::atable_ptr_t result = std::make_shared<MutableVerticalTable>(vc);
120  addResult(result);
121 }
