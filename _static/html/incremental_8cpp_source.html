18 void IncrementalCandidateLayouter::detectAffectedPartitions(Query *q) {
19  // Now get all containers and compare to the best original
20  // layout
21  Result currentBest = getBestResult();
22  Layout::internal_layout_t currentLayout = currentBest.layout.raw();
23 
24  // Create a set for easier deletion
25  set_subset_t current = set_subset_t(q->queryAttributes.begin(), q->queryAttributes.end());
26 
27  // Compare the current query set against all others
28  for (size_t j = 0; j < currentLayout.size(); ++j) {
29  subset_t base = currentLayout[j];
30 
31  bool found = false;
32  BOOST_FOREACH(unsigned baseAttr, base) {
33  BOOST_FOREACH(unsigned queryAttr, current) {
34  if (baseAttr == queryAttr) {
35  current.erase(queryAttr);
36  found = true;
37  break; // exit to container
38  }
39  }
40 
41  if (found) {
42  _splittedSets.push_back(base);
43  break; // exit to next container
44  }
45  }
46 
47  // Capture all unsused partitions
48  if (!found) {
49  _unusedPartitions.push_back(base);
50  }
51  }

96 Result IncrementalCandidateLayouter::rewriteResult(Result affectedBest) {

101

106  }
107 
108  rewritten.add(newContainer);

110 
111  // Add the unused partitions to the new result layout
112  BOOST_FOREACH(subset_t t, _unusedPartitions)
113  rewritten.add(t);
114 
115  // SInce we cached the cost for all queries (including the new
116  // one) we can now insert the modified cost at the slots where
117  // the cost are 0 because the query is not part of the cached
118  // ones
119  for (size_t i = 0, j = 0; i < _unusedCostForPartitions.size(); ++i) {
120  if (_unusedCostForPartitions[i] == 0.0) {
121  _unusedCostForPartitions[i] = affectedBest.cost[j++];
122  }
123  }
124

126  //Result newresult(rewritten, getCost(rewritten));
