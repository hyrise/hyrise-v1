52 void RadixCluster::executeClustering() {
53  const auto &tab = getInputTable();
54  auto tableSize = tab->size();
55  auto field = _field_definition[0];
56 
57  // Result Vector
58  const auto &result = getInputTable(1);
59 
60  // Get the prefix sum from the input
61  const auto &prefix_sum = getInputTable(2);
62  const auto &data_prefix_sum = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(getDataVector(prefix_sum).first->copy());
63 
64  // Prepare mask
65  auto mask = ((1 << bits()) - 1) << significantOffset();
66 
67  // Cast the vectors to the lowest part in the hierarchy
68  const auto &data_hash = getDataVector(result).first;
69  const auto &data_pos = getDataVector(result, 1).first;
70 
71  // Calculate start stop
72  _start = 0; _stop = tableSize;
73  if (_count > 0) {
74  _start = (tableSize / _count) * _part;
75  _stop = (_count -1) == _part ? tableSize : (tableSize/_count) * (_part + 1);
76  }
77 
78  // check if tab is PointerCalculator; if yes, get underlying table and actual rows and columns
79  auto p = std::dynamic_pointer_cast<const PointerCalculator>(tab);
80  if (p) {
81  auto ipair = getDataVector(p->getActualTable());
82  const auto &ivec = ipair.first;
83 
84  const auto &dict = std::dynamic_pointer_cast<OrderPreservingDictionary<T>>(tab->dictionaryAt(p->getTableColumnForColumn(field)));
85  const auto &offset = p->getTableColumnForColumn(field) + ipair.second;
86 
87  auto hasher = std::hash<T>();
88  for(decltype(tableSize) row = _start; row < _stop; ++row) {
89  // Calculate and increment the position
90  auto hash_value = hasher(dict->getValueForValueId(ivec->get(offset, p->getTableRowForRow(row))));//ts(tpe, fun);
91  auto offset = (hash_value & mask) >> _significantOffset;
92  auto pos_to_write = data_prefix_sum->inc(0, offset);
93 
94  // Perform the clustering
95  data_hash->set(0, pos_to_write, hash_value);
96  data_pos->set(0, pos_to_write, p->getTableRowForRow(row));
97  }
98  } else {
99  auto ipair = getDataVector(tab);
100  const auto &ivec = ipair.first;
101  const auto &dict = std::dynamic_pointer_cast<OrderPreservingDictionary<T>>(tab->dictionaryAt(field));
102  const auto &offset = field + ipair.second;
103 
104  std::hash<T> hasher;
105  for(decltype(tableSize) row = _start; row < _stop; ++row) {
106  // Calculate and increment the position
107  auto hash_value = hasher(dict->getValueForValueId(ivec->get(offset, row)));//ts(tpe, fun);
108  auto offset = (hash_value & mask) >> _significantOffset;
109  auto pos_to_write = data_prefix_sum->inc(0, offset);
110 
111  // Perform the clustering
112  data_hash->set(0, pos_to_write, hash_value);
113  data_pos->set(0, pos_to_write, row);
114  }
115  }
116  addResult(result);
117 }
