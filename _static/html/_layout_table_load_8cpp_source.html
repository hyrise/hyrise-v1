22 void LayoutTableLoad::executePlanOperation() {
23  const auto &input = this->input.getTable(0);
24  if ((input->columnCount() >= 1) && (input->size() >= _input_row)) {
25  std::string tmp_table_description = input->getValue<std::string>(0, _input_row);
26 
27  std::vector<std::string> description_lines;
28  splitString(description_lines, tmp_table_description, "\n");
29 
30  if (_override_group != "") {
31  std::vector<std::string> group_lines;
32  splitString(group_lines, description_lines[2], "|");
33  for (size_t index = 0; index < group_lines.size(); ++index) {
34  if (_override_group == "C")
35  group_lines[index] = std::to_string(index) + "_C";
36  else if (_override_group == "R")
37  group_lines[index] = "0_R";
38  else
39  throw std::runtime_error("unknown override_group " + _override_group);
40  }
41  description_lines[2] = joinString(group_lines, "|");
42  }
43 
44  std::string table_description = joinString(description_lines, "\n");
45 
46  StringHeader header(table_description);
47  CSVInput input(_file_name, CSVInput::params().setCSVParams(csv::HYRISE_FORMAT));
48  StorageManager *sm = StorageManager::getInstance();
49  sm->loadTable(_table_name, Loader::params().setHeader(header).setInput(input));
50  addResult(sm->getTable(_table_name));
51  } else {
52  throw std::runtime_error("Input of LayoutTableLoad doesn't have at least one field to begin with");
53  }
54 }
