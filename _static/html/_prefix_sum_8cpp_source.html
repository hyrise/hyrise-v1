16 // calculates the prefix Sum for a given table
17 void PrefixSum::executePlanOperation() {
18  // get attribute vector of input table
19  const auto &in = getInputTable();
20  const size_t table_size = in->size();
21 
22  // get attribute vector of output table
23  std::vector<const ColumnMetadata *> metadata;
24  metadata.push_back(in->metadataAt(0));
25  auto output = std::make_shared<Table<>>(&metadata, nullptr, table_size, true, 0, 0, false);
26  output->resize(table_size);
27  const auto &oavs = output->getAttributeVectors(0);
28  auto ovector = std::dynamic_pointer_cast<FixedLengthVector<value_id_t>>(oavs.at(0).attribute_vector);
29 
30  // Build ivector list to avoid lock contention while getting the vectors
31  const size_t ivec_size = input.numberOfTables();
32  std::vector<vec_ref_t> ivecs;
33  for(size_t i=0; i < input.numberOfTables(); ++i) {
34  ivecs.emplace_back(getDataVector(getInputTable(i)).first);
35  }
36 
37  // calculate the prefix sum based on the index and the number of inputs
38  // we need to look at to calculate the correct offset
39  value_id_t sum = 0;
40  for(size_t i=0; i < table_size; ++i) {
41  sum += sumForIndex(ivec_size, ivecs, i);
42  ovector->set(0, i, sum + sumForIndexPrev(ivec_size, ivecs, i));
43  }
44 
45  addResult(output);
46 }
