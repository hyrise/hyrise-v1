32 void TableGenerator::prepare(size_t p) {
33  if (p <= _prepareSize)
34  return;
35 
36  if (_prepareSize > 0) {
37  _values.clear();
38  }
39 
40  _prepareSize = p;
41 
42  boost::uniform_int<int64_t> dist(0, std::numeric_limits<int64_t>::max());
43 
44  // First, generate all random numbers we need, this is blazing
45  // fast
46  std::vector<int64_t> v;
47 
48  // Depending on the cycle of the random number generator, we will
49  // need to generate more random numbers than we actually need to
50  // make sure we have still enough without double entries
51  int upper = (_prepareSize * 105) / 100;
52  for (int i = 0; i < upper; ++i) {
53  v.push_back(dist(gen));
54  }
55 
56  // Now, sort the vector
57  std::sort(v.begin(), v.end());
58  std::set<int64_t>::iterator it = _values.begin();
59 
60  // Insert the elements from the vector sorted to the set and ta da
61  // we are fast
62  for (size_t i = 0; i < v.size(); ++i) {
63  it = _values.insert(it, v[i]);
64  }
65 
66  // If we have not enough distinct values we will need to generate
67  // more random values until we are done, but this should be only a
68  // small amount.
69  while (_values.size() < _prepareSize) {
70  _values.insert(dist(gen));
71  }
72 
73 }

197 std::vector<int64_t> *TableGenerator::create_dicts(size_t dict_size_main, size_t dict_size_delta, size_t intersection, size_t tail) {
198  Progress p(dict_size_main + dict_size_delta);
199 
200  std::vector<int64_t> *v = new std::vector<int64_t>;
201  v->reserve(dict_size_main + dict_size_delta);
202 
203  // maindict
204  for (size_t i = 0; i < dict_size_main; ++i) {
205  v->push_back(dict_size_delta + i);
206  p.tick();
207  }
208 
209  // deltadict
210  size_t head = dict_size_delta - intersection - tail;
211 
212  for (size_t i = 0; i < head; ++i) {
213  v->push_back(i);
214  p.tick();
215  }
216 
217  for (size_t i = head; i < head + intersection; ++i) {
218  v->push_back(v->at(i - head));
219  p.tick();
220  }
221 
222  for (size_t i = head + intersection; i < dict_size_delta; ++i) {
223  v->push_back(dict_size_main + dict_size_delta + i);
224  p.tick();
225  }
226 
227  v->at(dict_size_main + dict_size_delta - tail - 1) = v->at(dict_size_main - 1);
228
