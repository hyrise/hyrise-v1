168 template<>
169 inline bool ConfigFile::string_as_T<bool>(const std::string &s) {
170  // Convert from a std::string to a bool
171  // Interpret "false", "F", "no", "n", "0" as false
172  // Interpret "true", "T", "yes", "y", "1", "-1", or anything else as true
173  bool b = true;
174  std::string sup = s;
175 
176  for (std::string::iterator p = sup.begin(); p != sup.end(); ++p) {
177  *p = toupper(*p); // make std::string all caps
178  }
179 
180  if (sup == std::string("FALSE") || sup == std::string("F") ||
181  sup == std::string("NO") || sup == std::string("N") ||
182  sup == std::string("0") || sup == std::string("NONE")) {
183  b = false;
184  }
185 
186  return b;
187 }

190 template<class T>
191 T ConfigFile::read(const std::string &key) const {
192  // Read the value corresponding to key
193  mapci p = myContents.find(key);
194 
195  if (p == myContents.end()) {

199  return string_as_T<T>(p->second);

217 template<class T>
218 bool ConfigFile::readInto(T &var, const std::string &key) const {
219  // Get the value corresponding to key and store in var
220  // Return true if key is found
221  // Otherwise leave var untouched
222  mapci p = myContents.find(key);
223  bool found = (p != myContents.end());
224 
225  if (found) {
226  var = string_as_T<T>(p->second);
227  }
228 
229  return found;
230 }

233 template<class T>
234 bool ConfigFile::readInto(T &var, const std::string &key, const T &value) const {
235  // Get the value corresponding to key and store in var
236  // Return true if key is found
237  // Otherwise set var to given default
238  mapci p = myContents.find(key);
239  bool found = (p != myContents.end());
240 
241  if (found) {
242  var = string_as_T<T>(p->second);
243  } else {
244  var = value;
245  }
246 
247  return found;
248 }
