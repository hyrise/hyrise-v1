81 template<typename T>
82 adj_t Matrix<T>::buildAdjacency() const {
83  adj_t result;
84  result.xadj = (int *) malloc((numVertices() + 1) * sizeof(*result.xadj));
85  result.adjncy = (int *) malloc(2 * numEdges() * sizeof(*result.adjncy));
86  result.adjwgt = (int *) malloc(2 * numEdges() * sizeof(*result.adjwgt));
87 
88  // Now write the output, for each vertice we have to write all edges
89  int currentAdjxPos = 0;
90  int currentAdjxOffset = 0;
91 
92  int currentAdjncyOffset = 0;
93  for (int i = 0; i < numVertices(); ++i) {
94  // For each vertex check all possible other vertices for connections
95  for (int j = 0; j < _numCols; ++j) {
96  if (i == j)
97  continue;
98 
99  int value = get(i, j);
100  if (value > 0) {
101  result.adjncy[currentAdjncyOffset] = j;
102  result.adjwgt[currentAdjncyOffset++] = value;
103  }
104  }
105 
106  result.xadj[currentAdjxPos++] = currentAdjxOffset;
107  currentAdjxOffset = currentAdjncyOffset;
108  }
109  result.xadj[currentAdjxPos] = currentAdjncyOffset;
110  return result;
111 }
