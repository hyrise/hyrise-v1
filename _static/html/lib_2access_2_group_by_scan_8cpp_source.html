153 template<typename HashTableType, typename MapType, typename KeyType>
154 void GroupByScan::executeGroupBy() {
155  auto resultTab = createResultTableLayout();
156  auto groupResults = getInputHashTable();
157  // Allocate some memory for the result tab and resize the table
158  resultTab->resize(groupResults->numKeys());
159 
160  pos_t row = 0;
161  typename HashTableType::map_const_iterator_t it1, it2, end;
162  // set iterators: in the sequential case, getInputTable() returns an AggregateHashTable, in the parallel case a HashTableView<>
163  // Alternatively, a common type could be introduced
164  if (_count < 1) {
165  auto aggregateHashTable = std::dynamic_pointer_cast<const HashTableType>(groupResults);
166  it1 = aggregateHashTable->getMapBegin();
167  end = aggregateHashTable->getMapEnd();
168  } else {
169  auto hashTableView = std::dynamic_pointer_cast<const HashTableView<MapType, KeyType> >(groupResults);
170  it1 = hashTableView->getMapBegin();
171  end = hashTableView->getMapEnd();
172  }
173  for (it2 = it1; it1 != end; it1 = it2) {
174  // outer loop over unique keys
175  auto pos_list = std::make_shared<pos_list_t>();
176  for (; (it2 != end) && (it1->first == it2->first); ++it2) {
177  // inner loop, all keys equal to it1->first
178  pos_list->push_back(it2->second);
179  }
180  writeGroupResult(resultTab, pos_list, row);
181  row++;
182  }
183  this->addResult(resultTab);
184 }
