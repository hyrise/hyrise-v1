170 double Query::hyriseEquivalentProjection(bool isSelective) {
171 
172  // compute the offset for each attribute
173  int offset = 0;
174  std::vector<int> attsOffset(_container.size());
175 
176  for (unsigned i = 0; i < _container.size(); i++) {
177  attsOffset[i] = offset;
178  offset += _schema.attributes[_container[i]];
179  }
180 
181  int containerWidth = offset;
182 
183  // start with the first attribute and try to make equivalent projection
184  std::vector<int> eqProjectionOffset;
185  std::vector<int> eqProjectionEnd;
186 
187  int currentProjection = -1;
188 
189  for (unsigned i = 0; i < _container.size(); ++i) {
190  // i is the position of the att in the container
191  // att is the attribute number corresponding to i
192  int att = _container[i];
193 
194  // if the attribute is projected, add it to projection
195  if (std::find(queryAttributes.begin(), queryAttributes.end(), att) == queryAttributes.end())
196  // att is not projected
197  {
198  continue;
199  } else {
200  if (currentProjection == -1) {
201  // first projection, add it
202  currentProjection = 0;
203  eqProjectionOffset.push_back(attsOffset[i]);
204  eqProjectionEnd.push_back(attsOffset[i] + _schema.attributes[att]);
205  continue;
206  }
207 
208  // new projected attribute; check if the gap is big
209  int gap = attsOffset[i] - eqProjectionEnd[currentProjection];
210 
211  if (gap < CACHE_LINE_SIZE) {
212  // same projection
213  eqProjectionEnd[currentProjection] = attsOffset[i] + _schema.attributes[att];
214  } else {
215  // new projection
216  currentProjection++;
217  eqProjectionOffset.push_back(attsOffset[i]);
218  eqProjectionEnd.push_back(attsOffset[i] + _schema.attributes[att]);
219  }
220  }
221  }
222 
223  //compute the gap between the first and last projection
224  if (currentProjection > 0) {
225  int interRowGap = eqProjectionOffset[0] + containerWidth - eqProjectionEnd[currentProjection];
226 
227  if (interRowGap < CACHE_LINE_SIZE) {
228  // merge last with first projection
229  eqProjectionOffset[0] = eqProjectionOffset[currentProjection];
230  eqProjectionEnd[0] = eqProjectionEnd[0] + containerWidth;
231 
232  eqProjectionOffset.pop_back();
233  eqProjectionEnd.pop_back();
234  }
235 
236  }
237 
238  double misses = 0;
239 
240  for (unsigned i = 0; i < eqProjectionOffset.size(); i++) {
241  int po = eqProjectionOffset[i];
242  int pw = eqProjectionEnd[i] - po;
243 
244  if (!isSelective) {
245  misses += hyrisePartialProjection(po, pw);
246  } else {
247  misses += hyriseSingleOOO(po, pw);
248  }
249  }
250 
251  return misses;
252 }
