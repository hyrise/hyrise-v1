117 void SimpleTableDump::dumpDictionary(std::string name, std::shared_ptr<AbstractTable> table, size_t col) {
118  std::string fullPath = _baseDirectory + "/" + name + "/" + table->nameOfColumn(col) + ".dict.dat";
119  std::ofstream data (fullPath, std::ios::out | std::ios::binary);
120 
121  // We make a small hack here, first we obtain the size of the
122  // dictionary then we virtually create all value ids, this can break
123  // if the dictionary has no contigous value ids
124  size_t dictionarySize = table->dictionaryAt(col)->size();
125  write_to_stream_functor fun(data, table);
126  type_switch<hyrise_basic_types> ts;
127  for(size_t i=0; i < dictionarySize; ++i) {
128  fun.setCol(col);
129  fun.setVid(i);
130  ts(table->typeOfColumn(col), fun);
131  }
132  data.close();
133 }

146 void SimpleTableDump::dumpHeader(std::string name, std::shared_ptr<AbstractTable> table) {
147  std::stringstream header;
148  std::vector<std::string> names, types;
149  std::vector<uint32_t> parts;
150 
151  // Get names and types
152  for(size_t i=0; i < table->columnCount(); ++i) {
153  names.push_back(table->nameOfColumn(i));
154  types.push_back(data_type_to_string(table->typeOfColumn(i)));
155  }
156 
157  // This calculation will break if the width of the value_id changes
158  // or someone forgets to simply update the width accordingly in the
159  // constructor of the table
160  for(size_t i=0; i < table->sliceCount(); ++i) {
161  parts.push_back(table->getSliceWidth(i) / sizeof(value_id_t));
162  }
163 
164  // Dump and join
165  header << std::accumulate(names.begin(), names.end(), std::string(), infix(" | ")) << "\n";
166  header << std::accumulate(types.begin(), types.end(), std::string(), infix(" | ")) << "\n";
167  std::vector<std::string> allParts;
168  for(size_t i=0; i < parts.size(); ++i) {
169  auto p = parts[i];
170  auto tmp = std::vector<std::string>(p, std::to_string(i) + "_R");
171  allParts.insert(allParts.end(), tmp.begin(), tmp.end());
172  }
173  header << std::accumulate(allParts.begin(), allParts.end(), std::string(), infix(" | ")) << "\n";
174  header << "===";
175 
176  std::string fullPath = _baseDirectory + "/" + name + "/header.dat";
177  std::ofstream data (fullPath, std::ios::out | std::ios::binary);
178  data << header.str();
179  data.close();
180 }

197 bool SimpleTableDump::dump(std::string name, std::shared_ptr<AbstractTable> table) {
198  verify(table);
199  auto mainTable = std::dynamic_pointer_cast<Store>(table)->getMainTables()[0];
200  prepare(name);
201  for(size_t i=0; i < mainTable->columnCount(); ++i) {
202  // For each attribute dump dictionary and values
203  dumpDictionary(name, mainTable, i);
204  dumpAttribute(name, mainTable, i);
205  }

209
