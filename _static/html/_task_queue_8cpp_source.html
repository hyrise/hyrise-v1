24 void TaskQueue::launchThread(int core) {
25  static hwloc_topology_t topology;;
26  if (topology == nullptr) {
27  hwloc_topology_init(&topology);
28  hwloc_topology_load(topology);
29  }
30 
31  //get the number of cores on system
32  static int NUM_PROCS = hwloc_get_nbobjs_by_type(topology, HWLOC_OBJ_CORE);
33  if (core < NUM_PROCS) {
34  _thread = new std::thread(&TaskQueue::executeTask, this);
35  hwloc_cpuset_t cpuset;
36  hwloc_obj_t obj;
37 
38  obj = hwloc_get_obj_by_type(topology, HWLOC_OBJ_CORE, core);
39  // the bitmap to modify
40  cpuset = hwloc_bitmap_dup(obj->cpuset);
41  // remove hyperthreads
42  hwloc_bitmap_singlify(cpuset);
43  // bind
44  if (hwloc_set_thread_cpubind(topology, _thread->native_handle(), cpuset, HWLOC_CPUBIND_STRICT | HWLOC_CPUBIND_NOMEMBIND)) {
45  char *str;
46  int error = errno;
47  hwloc_bitmap_asprintf(&str, obj->cpuset);
48  printf("Couldn't bind to cpuset %s: %s\n", str, strerror(error));
49  free(str);
50  throw std::runtime_error(strerror(error));
51  }
52 
53  hwloc_bitmap_free(cpuset);
54 
55  } else {
56  // this case should never happen, as TaskQueue is only initialized from SimpleTaskScheduler, which captures this case
57  throw std::logic_error("CPU to run thread on is larger than number of total cores; seems that TaskQueue was initialized outside of SimpleTaskScheduler, which should not happen");
58  }
59 }

61 void TaskQueue::executeTask() {
62 
63 
64 
65 
66 
67 
68  //infinite thread loop
69  while (1) {
70 
71  //block protected by _threadStatusMutex, stop thread if about to stop
72  {
73  std::lock_guard<std::mutex> lk1(_threadStatusMutex);
74  if (_status == TO_STOP)
75  return;
76  }
77 
78  std::unique_lock<std::mutex> ul(_queueMutex);
79  // get task and execute
80  if (_runQueue.size() > 0) {
81  _blocked = true;
82  std::shared_ptr<Task> task = _runQueue.front();
83  _runQueue.pop();
84  ul.unlock();
85  if (task) {
86  LOG4CXX_DEBUG(logger, "Started executing task" << std::hex << &task << std::dec << " on core " << _core);
87  (*task)();
88  LOG4CXX_DEBUG(logger, "Executed task " << std::hex << &task << std::dec << " on core " << _core);
89  task->notifyDoneObservers();
90  }
91  }
92  // TODO wait until new task/steal work
93  else {
94  {
95  std::lock_guard<std::mutex> lk1(_threadStatusMutex);
96  if (_status == RUN_UNTIL_DONE || _status == TO_STOP)
97  return;
98  }
99  _blocked = false;
100  _condition.wait(ul);
101  }
102  }
103 }

116 std::queue<std::shared_ptr<Task> > *TaskQueue::stopQueue() {
117 
118  if (_status != STOPPED) {
119  _threadStatusMutex.lock();
120  // the thread to be stopped is either executing a task, or waits for the condition variable
121  // set status to "TO_STOP" so that the thread either quits after executing the task, or after having been notified by the condition variable
122  _status = TO_STOP;
123  //wake up thread in case thread is sleeping
124  _condition.notify_one();
125  _threadStatusMutex.unlock();
126  _thread->join();
127  delete _thread;
128  _thread = NULL;
129  _status = STOPPED;
130  return &_runQueue;
131  } else
132  // queue already stopped
133  return NULL;
134 }
