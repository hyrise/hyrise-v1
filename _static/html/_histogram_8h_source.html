66 template<typename T>
67 void Histogram::executeHistogram() {
68  const auto &tab = getInputTable();
69  const auto tableSize = getInputTable()->size();
70  const auto field = _field_definition[0];
71 
72  //Prepare mask
73  auto mask = ((1 << bits()) - 1) << significantOffset();
74 
75  // Prepare Output Table
76  auto result = createOutputTable(1 << bits());
77  auto pair = getDataVector(result);
78 
79  // Iterate and hash based on the part description
80  size_t start=0, stop=tableSize;
81  if (_count > 0) {
82  start = (tableSize / _count) * _part;
83  stop = (_count -1) == _part ? tableSize : (tableSize/_count) * (_part + 1);
84  }
85  // check if tab is PointerCalculator; if yes, get underlying table and actual rows and columns
86  auto p = std::dynamic_pointer_cast<const PointerCalculator>(tab);
87  if (p) {
88  auto ipair = getDataVector(p->getActualTable(), p->getTableColumnForColumn(field));
89  const auto &ivec = ipair.first;
90  const auto &dict = std::dynamic_pointer_cast<OrderPreservingDictionary<T>>(tab->dictionaryAt(p->getTableColumnForColumn(field)));
91  const auto &offset = ipair.second;
92 
93  auto hasher = std::hash<T>();
94  for(size_t row = start; row < stop; ++row) {
95  auto hash_value = hasher(dict->getValueForValueId(ivec->get(offset, p->getTableRowForRow(row))));
96  pair.first->inc(0, (hash_value & mask) >> significantOffset());
97  }
98  } else {
99  auto ipair = getDataVector(tab, field);
100  const auto &ivec = ipair.first;
101  const auto &dict = std::dynamic_pointer_cast<OrderPreservingDictionary<T>>(tab->dictionaryAt(field));
102  const auto &offset = ipair.second;
103 
104  auto hasher = std::hash<T>();
105  for(size_t row = start; row < stop; ++row) {
106  auto hash_value = hasher(dict->getValueForValueId(ivec->get(offset, row)));
107  pair.first->inc(0, (hash_value & mask) >> significantOffset());
108  }
109  }
110  addResult(result);
111 }
