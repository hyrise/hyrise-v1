51  std::vector<pos_t>* sort() const {
52  std::vector<pair_t> result;
53 
54  for (size_t row = 0; row < _t->size(); ++row) {
55  result.push_back({ExtractFunctor<T>::extractValue(_t, _f, row), row});
56  }
57 
58  std::stable_sort(result.begin(),
59  result.end(),
60  [](const pair_t& left, const pair_t& right)
61  { return left.value < right.value; });
62 
63  auto r = new std::vector<pos_t>;
64  r->reserve(result.size());
65  for (const pair_t& p: result)
66  r->push_back(p.row);
67 
68  return r;
69  }

79 void SortScan::executePlanOperation() {
80  const auto& table = input.getTable(0);
81  // Sorted Position List
82  std::vector<pos_t> *sorted_pos;
83 
84  // When table is not only a table but also using an ordered dictionary on sort field,
85  // we can just sort by value_id
86  // TODO: fix Table<> template
87  auto base_table = std::dynamic_pointer_cast<const Table<>>(table);
88  if ((table->dictionaryAt(_sort_field)->isOrdered()) && (base_table)) {
89  sorted_pos = ColumnSorter<ValueId, ExtractValueId>(table, _sort_field).sort();
90  } else {
91  switch (table->metadataAt(_sort_field)->getType()) {
92  case IntegerType:
93  sorted_pos = ColumnSorter<hyrise_int_t, ExtractValue>(table, _sort_field).sort();
94  break;
95  case FloatType:
96  sorted_pos = ColumnSorter<hyrise_float_t, ExtractValue>(table, _sort_field).sort();
97  break;
98  case StringType:
99  sorted_pos = ColumnSorter<hyrise_string_t, ExtractValue>(table, _sort_field).sort();
100  break;
101  default:
102  throw std::runtime_error("Datatype not supported");
103  }
104  }
105 
106  storage::atable_ptr_t result;
107 
108  if (producesPositions) {
109  result = PointerCalculatorFactory::createPointerCalculatorNonRef(table, nullptr, sorted_pos);
110  } else {
111  result = table->copy_structure_modifiable(nullptr, true);
112  size_t result_row = 0;
113  for (const auto& p: *sorted_pos) {
114  result->copyRowFrom(table, p, result_row++);
115  }
116  delete sorted_pos;
117  }
118  addResult(result);
119 }
