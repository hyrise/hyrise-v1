81 template<class T>
82 class GroupKeyHash {
83 public:
84  size_t operator()(const T &key) const {
85  static auto hasher = std::hash<value_id_t>();
86 
87  std::size_t seed = 0;
88  for (size_t i = 0, key_size = key.size();
89  i < key_size; ++i) {
90  // compare boost hash_combine
91  seed ^= hasher(key[i]) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
92  }
93  return seed;
94  }

96  static T getGroupKey(const hyrise::storage::c_atable_ptr_t &table,
97  const field_list_t &columns,
98  const size_t fieldCount,
99  const pos_t row) {
100  ValueIdList value_list = table->copyValueIds(row, &columns);
101  T key;
102  for (size_t i = 0, key_size = fieldCount; i < key_size; i++)
103  key.push_back(extract<T>(table, columns[i], value_list[i]));
104  return key;
105  }

170  _dirty = true;
