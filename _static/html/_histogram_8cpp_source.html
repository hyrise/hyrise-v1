90 void Histogram2ndPass::executePlanOperation() {
91  const auto &tab = getInputTable();
92  const auto tableSize = getInputTable()->size();
93  const auto field = 0;//_field_definition[0];
94 
95  //Prepare mask
96  auto mask = ((1 << bits()) - 1) << significantOffset();
97  auto mask2 = (( 1 << _bits2 ) - 1) << _significantOffset2;
98 
99  // Prepare Output Table
100  auto result = createOutputTable((1<<_bits2) * (1 << _bits));
101  auto o_data = getDataVector(result).first;
102 
103  // Get input vector
104  auto i_data = getDataVector(tab).first;
105 
106  // Iterate and hash based on the part description
107  size_t start=0, stop=tableSize;
108  if (_count > 0) {
109  start = (tableSize / _count) * _part;
110  stop = (_count -1) == _part ? tableSize : (tableSize/_count) * (_part + 1);
111  }
112  for(size_t row = start; row < stop; ++row) {
113  auto hash_value = i_data->get(field, row);
114  auto offset = (hash_value & mask) * (1<<_bits2) + ((hash_value & mask2) >> _significantOffset2);
115  o_data->inc(0, offset);
116  }
117 
118  addResult(result);
119 }
