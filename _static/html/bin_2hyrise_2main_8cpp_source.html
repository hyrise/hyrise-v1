44  assert((start < end) && "start must be smaller than end");

96 void bindToNode(int node) {
97  hwloc_topology_t topology = getHWTopology();
98  hwloc_cpuset_t cpuset;
99  hwloc_obj_t obj;
100 
101  // The actual core
102  obj = hwloc_get_obj_by_type(topology, HWLOC_OBJ_CORE, node);
103  cpuset = hwloc_bitmap_dup(obj->cpuset);
104  hwloc_bitmap_singlify(cpuset);
105 
106  // bind
107  if (hwloc_set_cpubind(topology, cpuset, HWLOC_CPUBIND_STRICT | HWLOC_CPUBIND_NOMEMBIND | HWLOC_CPUBIND_PROCESS)) {
108  char *str;
109  int error = errno;
110  hwloc_bitmap_asprintf(&str, obj->cpuset);
111  printf("Couldn't bind to cpuset %s: %s\n", str, strerror(error));
112  free(str);
113  throw std::runtime_error(strerror(error));
114  }
115 
116  // free duplicated cpuset
117  hwloc_bitmap_free(cpuset);
118 
119 
120  obj = hwloc_get_obj_by_type(topology, HWLOC_OBJ_MACHINE, node);
121  if (hwloc_set_membind_nodeset(topology, obj->nodeset, HWLOC_MEMBIND_INTERLEAVE, HWLOC_MEMBIND_STRICT | HWLOC_MEMBIND_THREAD)) {
122  char *str;
123  int error = errno;
124  hwloc_bitmap_asprintf(&str, obj->nodeset);
125  fprintf(stderr, "Couldn't membind to nodeset %s: %s\n", str, strerror(error));
126  fprintf(stderr, "Continuing as normal, however, no guarantees\n");
127  free(str);
128  //throw std::runtime_error(strerror(error));
